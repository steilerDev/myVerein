\newenvironment{RESTTable}[2]{
	% total width = 1 - number of columns * 2 * \tabcolsep (above)
	\begin{longtable}{p{.20\textwidth} | p{.1\textwidth} | p{.22\textwidth} | p{.36\textwidth}}
    	\caption{#1} \label{#2} \\
			\textbf{URI} & \textbf{Request method} & \textbf{Parameters} & \textbf{Description} \\
		\endfirsthead
			\multicolumn{4}{c}
			{{\bfseries \tablename\ \thetable{} -- continued from previous page}} \\
			\textbf{URI} & \textbf{Request method} & \textbf{Parameters} & \textbf{Description} \\
		\endhead
			\multicolumn{4}{c}{{Continued on next page}} \\ 
		\endfoot
		\endlastfoot
}
{
	\end{longtable}
}
